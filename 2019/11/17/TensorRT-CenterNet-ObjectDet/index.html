def forward(self, x):
x = self.base(x)
x = self.dla_up(x[self.first_level:])

# x = self.fc(x)
# y = self.softmax(self.up(x))
ret = []
for head in self.heads:
    ret.append(self.__getattr__(head)(x))
sigmoid = ret[0].sigmoid_()
hmax = nn.functional.max_pool2d(
    sigmoid, (3, 3), stride=1, padding=1)
ret[0] = hmax
ret.append(sigmoid)
return [ret]
